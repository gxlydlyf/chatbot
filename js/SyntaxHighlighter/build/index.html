Hello SyntaxHighlighter

var code = 'YOUR_CODE_HERE'; // 替换成你的代码字符串
var highlightedCode = code.replace(/&/g, '&')
                         .replace(/< /g, '<')
                         .replace(/>/g, '>');

var preElem = document.createElement('pre');
var codeElem = document.createElement('code');
codeElem.innerHTML = highlightedCode;

preElem.appendChild(codeElem);
document.body.appendChild(preElem);
